class CircularArc:
    """
    Create a NURBS representation of a circular arc

    Create a NURBS representation of the circular arc parametrized by
        C(u) = O + R*cos(u)*X + R*sin(u)*Y
        theta_start <= u <= theta_end

    Parameters
    ----------

    O : ndarray with shape (ndim,)
        Center of the circular arc. ndim=2 or ndim=3

    X : ndarray with shape (ndim,)
        Unit vector lying in the plane of the circle

    Y : ndarray with shape (ndim,)
        Unit vector lying in the plane of the circle and orthogonal to X

    R : scalar
        Radius of the circle

    theta_start : scalar
        Start angle, measured with respect to X

    theta_end : scalar
        End angle, measured with respect to X

    References
    ----------
    The NURBS book. Chapter 7.5
    L. Piegl and W. Tiller
    Springer, second edition

    """
    def __init__(self, O, X, Y, R, theta_start, theta_end):
        # Convert all inputs to JAX arrays
        self.O = jnp.asarray(O)
        self.X = jnp.asarray(X)
        self.Y = jnp.asarray(Y)
        self.R = jnp.asarray(R)
        self.theta_start = jnp.asarray(theta_start)
        self.theta_end = jnp.asarray(theta_end)

        # Normalize direction vectors
        self.X = self.X / jnp.linalg.norm(self.X)
        self.Y = self.Y / jnp.linalg.norm(self.Y)

        # Check dimensionality
        self.ndim = self.O.shape[0]
        if self.ndim not in (2, 3):
            raise ValueError("The number of dimensions must be 2 or 3.")

        # Create the NURBS representation
        self.make_nurbs_circular_arc()

    def make_nurbs_circular_arc(self):
        # Rename variables for brevity
        O, X, Y, R = self.O, self.X, self.Y, self.R
        theta_start, theta_end = self.theta_start, self.theta_end
        ndim = self.ndim

        if theta_end < theta_start:
            theta_end = theta_end + 2 * jnp.pi

        theta = theta_end - theta_start

        # Determine number of arc segments (max 4 for full circle)
        n_arcs = jnp.where(theta <= 0.5 * jnp.pi, 1,
                   jnp.where(theta <= 1.0 * jnp.pi, 2,
                   jnp.where(theta <= 1.5 * jnp.pi, 3,
                   jnp.where(theta <= 2.0 * jnp.pi, 4, -1)))).item()

        if n_arcs == -1:
            raise ValueError("Invalid circular arc angle span.")

        # Angle spanned by each segment
        delta_theta = theta / n_arcs

        # Initialize arrays of control points and weights
        n = 2 * n_arcs
        P = jnp.zeros((ndim, n + 1))
        W = jnp.zeros((n + 1,))

        # First control point
        P0 = O + R * jnp.cos(theta_start) * X + R * jnp.sin(theta_start) * Y
        T0 = -jnp.sin(theta_start) * X + jnp.cos(theta_start) * Y
        P = P.at[:, 0].set(P0)
        W = W.at[0].set(1.0)

        # Get the coordinates and weights of the other control points
        index, angle = 0, theta_start
        for _ in range(n_arcs):
            # Angle spanned by the current segment
            angle = angle + delta_theta

            # End point of the current segment
            P2 = O + R * jnp.cos(angle) * X + R * jnp.sin(angle) * Y
            T2 = -jnp.sin(angle) * X + jnp.cos(angle) * Y

            # Solve the intersection between tangent lines T0 and T2 to compute the intermediate point
            P1 = self.intersect_lines(P0, T0, P2, T2)
            W1 = jnp.cos(delta_theta / 2)

            # Store control points and weights
            P = P.at[:, index + 1].set(P1)
            P = P.at[:, index + 2].set(P2)
            W = W.at[index + 1].set(W1)
            W = W.at[index + 2].set(1.0)

            # Get ready for the next segment!
            index, P0, T0 = index + 2, P2, T2

        # Define the order and knot vector of the basis polynomials
        p = 2
        U = jnp.zeros((n + p + 2,))
        U = U.at[0:3].set(0.0)
        U = U.at[-3:].set(1.0)

         # Set the multiplicity of the interior knots to 2 to connect the segments
        if n_arcs == 2:
            U = U.at[3:5].set(0.5)
        elif n_arcs == 3:
            U = U.at[3:5].set(1 / 3)
            U = U.at[5:7].set(2 / 3)
        elif n_arcs == 4:
            U = U.at[3:5].set(1 / 4)
            U = U.at[5:7].set(2 / 4)
            U = U.at[7:9].set(3 / 4)

        # Create the NURBS curve
        self.P, self.W, self.p, self.U = P, W, p, U
        self.NurbsCurve = NurbsCurve(P, W, p, U)

    @staticmethod
    def intersect_lines(P0, T0, P2, T2):
        """Compute the intersection after reducing the 3x2 system to a 2x2 system using the dot product"""
        A = jnp.array([[jnp.dot(T0, T0), -jnp.dot(T0, T2)],
                       [jnp.dot(T0, T2), -jnp.dot(T2, T2)]])
        b = jnp.array([jnp.dot(P2, T0) - jnp.dot(P0, T0),
                       jnp.dot(P2, T2) - jnp.dot(P0, T2)])
        u, v = jnp.linalg.solve(A, b)
        P1 = P0 + u * T0
        return P1

    def plot(self):

        # Plot the NURBS curve
        fig, ax = self.NurbsCurve.plot()

        # Plot additional details (full circle, circle center and start/end points)
        if self.ndim == 2:
            u = jnp.linspace(0, 2 * jnp.pi, 100)
            P_start = jnp.real(self.NurbsCurve.P[:, 0])
            P_end = jnp.real(self.NurbsCurve.P[:, -1])
            O = jnp.real(self.O[:, jnp.newaxis])
            X = jnp.real(self.X[:, jnp.newaxis])
            Y = jnp.real(self.Y[:, jnp.newaxis])
            R = jnp.real(self.R)
            x, y = O + R * jnp.cos(u) * X + R * jnp.sin(u) * Y
            ax.plot(x, y, 'b')
            ax.plot([0, self.O[0]], [0, self.O[1]], 'ko-', markersize=3.5, markerfacecolor='w')
            ax.plot([self.O[0], P_start[0]], [self.O[1], P_start[1]], 'ko-', markersize=3.5, markerfacecolor='w')
            ax.plot([self.O[0], P_end[0]], [self.O[1], P_end[1]], 'ko-', markersize=3.5, markerfacecolor='w')

        elif self.ndim == 3:

            # Plot the full circle
            u = jnp.linspace(0, 2*jnp.pi, 100)
            P_start = jnp.real(self.NurbsCurve.P[:, 0])
            P_end = jnp.real(self.NurbsCurve.P[:, -1])
            O = jnp.real(self.O[:, jnp.newaxis])
            X = jnp.real(self.X[:, jnp.newaxis])
            Y = jnp.real(self.Y[:, jnp.newaxis])
            R = jnp.real(self.R)
            x, y, z =  O + R*jnp.cos(u)*X + R*jnp.sin(u)*Y
            ax.plot(x, y, z, 'b')
            ax.plot([0, self.O[0]], [0, self.O[1]], [0, self.O[2]], 'ko-', markersize=3.5, markerfacecolor='w')
            ax.plot([self.O[0], P_start[0]], [self.O[1], P_start[1]], [self.O[2], P_start[2]], 'ko-', markersize=3.5, markerfacecolor='w')
            ax.plot([self.O[0], P_end[0]], [self.O[1], P_end[1]], [self.O[2], P_end[2]], 'ko-', markersize=3.5, markerfacecolor='w')

        else:
            raise Exception('The number of dimensions must be 2 or 3')


        # Rescale the plot
        self.NurbsCurve.rescale_plot(fig, ax)

        return fig, ax
